(require '[clojure.test])
(defmacro check
  [value & [pred & args]]
  (let [is-form (if pred
                  `(~pred ~value ~@args)
                  value)
        check-name (-> (gensym "check")
                       (vary-meta assoc
                                  :test `(fn [] (clojure.test/is ~is-form))
                                  :kindly/kind :kind/check))]
    `(def ~check-name
       {:value ~value
        :check ~is-form})))
(check false)
